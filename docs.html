# add a doc with nav

  1. add doc file config.md in your docs

  2. add nav /config for your doc file config.md

  //nav/en.js
  {
    text: 'Config',
    link: '/config'
  }
  //nav/zh.js
  {
    text: '配置',
    link: '/config'
  }

# add some docs with nav

  1. add doc dir demos in your docs

  2. add nav /demos/ for your doc files demos/*.md

  //nav/en.js
  {
    text: 'Demos',
    link: '/demos'
  }
  //nav/zh.js
  {
    text: '示例',
    link: '/demos'
  }

# add a doc without nav in a sub url

  1. add doc file xx.md in your docs demos/ path

  2. has sidebar? add sidebar xx for your doc files demos/xx.md

  //sidebar/demos.js
  [
  '',//
  'xx'// for xx.md
